######Data Graph

CREATE (N_1:number {number:'1'})
CREATE (N_2:number {number:'2'})
CREATE (N_3:number {number:'3'})
CREATE (N_4:number {number:'4'})
CREATE
  (N_2)-[:forward]->(N_3),
  (N_1)-[:forward]->(N_4),
  (N_1)-[:forward]->(N_2),
  (N_1)-[:forward]->(N_3)


######Query Graph
CREATE (N_a:letter {letter:'a'})
CREATE (N_b:letter {letter:'b'})
CREATE (N_c:letter {letter:'c'})
CREATE
  (N_a)-[:forward]->(N_b),
  (N_a)-[:forward]->(N_c)




####### delete all graphs
MATCH (n) DETACH DELETE n

####### show graph
MATCH (n) RETURN n
